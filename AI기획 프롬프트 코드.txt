  <script>
    // Web Audio API 기반 오디오 엔진 (기계식 청축 사운드 합성기)
    let audioCtx = null;
    let lofiInterval = null;
    let isPlayingLofi = false;

    function initAudio() {
      if (!audioCtx) {
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
      }
    }

    // 실제 청축(Blue Switch) 고유의 가볍고 카랑카랑한 클릭음 합성 알고리즘
    function playBlueSwitch() {
      try {
        initAudio();
        if (audioCtx.state === 'suspended') {
          audioCtx.resume();
        }

        const now = audioCtx.currentTime;
        
        // 1. 하이 피치 금속편 스프링 튕기는 클릭 바 소리 (찰칵)
        const clickOsc = audioCtx.createOscillator();
        const clickGain = audioCtx.createGain();
        clickOsc.type = 'sine';
        clickOsc.frequency.setValueAtTime(2600, now);
        clickOsc.frequency.exponentialRampToValueAtTime(1500, now + 0.01);
        
        clickGain.gain.setValueAtTime(0.06, now);
        clickGain.gain.exponentialRampToValueAtTime(0.001, now + 0.01);

        // 2. 보강판 타격 베이스 노이즈 (탁)
        const thudOsc = audioCtx.createOscillator();
        const thudGain = audioCtx.createGain();
        thudOsc.type = 'triangle';
        thudOsc.frequency.setValueAtTime(150, now);
        thudOsc.frequency.exponentialRampToValueAtTime(55, now + 0.05);

        thudGain.gain.setValueAtTime(0.25, now);
        thudGain.gain.exponentialRampToValueAtTime(0.001, now + 0.05);

        // 로우패스 필터로 플라스틱 키캡 통울림 하우징 질감 확보
        const filter = audioCtx.createBiquadFilter();
        filter.type = 'lowpass';
        filter.frequency.value = 1700;

        clickOsc.connect(clickGain);
        clickGain.connect(filter);

        thudOsc.connect(thudGain);
        thudGain.connect(filter);

        filter.connect(audioCtx.destination);

        clickOsc.start(now);
        clickOsc.stop(now + 0.015);
        thudOsc.start(now);
        thudOsc.stop(now + 0.06);
      } catch (e) {
        console.warn("오디오 플레이 장치 에러:", e);
      }
    }

    // Lofi 힐링 배경 멜로디 발생 장치
    function toggleLofiMusic() {
      initAudio();
      const btn = document.getElementById('musicBtn');
      const eqActive = document.getElementById('eqAnimation');
      const eqStatic = document.getElementById('eqStatic');

      if (isPlayingLofi) {
        isPlayingLofi = false;
        if (lofiInterval) {
          clearInterval(lofiInterval);
          lofiInterval = null;
        }
        btn.classList.remove('bg-blue-500', 'text-white');
        btn.classList.add('bg-white', 'text-slate-700');
        eqActive.classList.add('hidden');
        eqStatic.classList.remove('hidden');
        showToast("아지트의 백그라운드 Lofi 사운드를 일시정지합니다.");
      } else {
        if (audioCtx.state === 'suspended') {
          audioCtx.resume();
        }
        isPlayingLofi = true;
        btn.classList.add('bg-blue-500', 'text-white');
        btn.classList.remove('bg-white', 'text-slate-700');
        eqActive.classList.remove('hidden');
        eqStatic.classList.add('hidden');
        showToast("방 안에 흐르는 아늑한 레코드 풍의 Lofi 음악이 시작되었습니다.");

        playLofiLoop();
      }
    }

    function playLofiLoop() {
      if (!isPlayingLofi) return;

      const playPianoChord = () => {
        if (!isPlayingLofi) return;
        const now = audioCtx.currentTime;
        const chords = [
          [130.81, 164.81, 196.00, 246.94], // Cmaj7
          [110.00, 130.81, 164.81, 220.00], // Am7
          [146.83, 174.61, 220.00, 261.63], // Dm7
          [98.00, 146.83, 196.00, 246.94]   // G7
        ];
        const notes = chords[Math.floor(Math.random() * chords.length)];

        notes.forEach((freq, i) => {
          const osc = audioCtx.createOscillator();
          const gain = audioCtx.createGain();
          
          osc.type = 'sine';
          osc.frequency.setValueAtTime(freq, now + i * 0.15);
          
          // Wow & Flutter 효과 시뮬레이션 (아날로그 미세 진동)
          const vibrato = audioCtx.createOscillator();
          const vibratoGain = audioCtx.createGain();
          vibrato.frequency.setValueAtTime(5.0, now);
          vibratoGain.gain.setValueAtTime(0.8, now);
          vibrato.connect(vibratoGain);
          vibratoGain.connect(osc.frequency);

          gain.gain.setValueAtTime(0, now + i * 0.15);
          gain.gain.linearRampToValueAtTime(0.04, now + i * 0.15 + 0.5);
          gain.gain.exponentialRampToValueAtTime(0.0001, now + i * 0.15 + 3.0);

          const lp = audioCtx.createBiquadFilter();
          lp.type = 'lowpass';
          lp.frequency.setValueAtTime(700, now);

          osc.connect(lp);
          lp.connect(gain);
          gain.connect(audioCtx.destination);
          
          vibrato.start(now);
          osc.start(now + i * 0.15);
          osc.stop(now + 3.2);
          vibrato.stop(now + 3.2);
        });
      };

      playPianoChord();
      lofiInterval = setInterval(() => {
        if (!isPlayingLofi) return;
        playPianoChord();
      }, 4500);
    }

    // 범용 알림 토스트 발생기
    function showToast(message) {
      const container = document.getElementById('toastContainer');
      const toast = document.createElement('div');
      toast.className = "bg-slate-900/90 text-white text-xs px-4 py-2.5 rounded-full shadow-lg flex items-center gap-2 transform translate-y-2 opacity-0 transition-all duration-300 pointer-events-auto border border-white/10 font-sans-ui";
      toast.innerHTML = `
        <span class="w-1.5 h-1.5 rounded-full bg-blue-500 animate-ping"></span>
        <span>${message}</span>
      `;
      container.appendChild(toast);
      
      setTimeout(() => {
        toast.classList.remove('translate-y-2', 'opacity-0');
      }, 10);

      setTimeout(() => {
        toast.classList.add('translate-y-[-5px]', 'opacity-0');
        setTimeout(() => toast.remove(), 300);
      }, 2500);
    }

  <!-- 2D 키캡 자판 파티클 물리 렌더러 및 근접 센싱 시스템 스크립트 -->
  <script>
    const canvas = document.getElementById('globalParticleCanvas');
    const ctx = canvas.getContext('2d');
    let particles = [];

    // 가로 및 세로 뷰포트 크기 변경 완벽 대응
    function resizeCanvas() {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resizeCanvas);
    resizeCanvas();

    // 키캡 자판 블록 파티클 객체 정의 (실제 키캡 3D 형상 모델링)
    const keyLabels = ['Q', 'W', 'E', 'R', 'A', 'S', 'D', 'F', 'Shift', 'Enter', 'Alt', 'Ctrl', 'Esc', 'Space'];
    
    class KeycapParticle {
      constructor(x, y) {
        this.x = x;
        this.y = y;
        const angle = Math.random() * Math.PI * 2;
        const speed = Math.random() * 5 + 4;
        this.vx = Math.cos(angle) * speed;
        this.vy = Math.sin(angle) * speed - 3; // 초기 분사 궤도를 상향 설계
        this.opacity = 1;
        this.rotation = Math.random() * Math.PI * 2;
        this.rotationSpeed = (Math.random() - 0.5) * 0.12;
        this.label = keyLabels[Math.floor(Math.random() * keyLabels.length)];
        this.width = this.label.length > 3 ? 38 : 22;
        this.height = 20;
      }

      update() {
        this.x += this.vx;
        this.y += this.vy;
        this.vy += 0.18; // 가벼운 중력 물리 가속
        this.rotation += this.rotationSpeed;
        this.opacity -= 0.012; // 페이드 아웃 타임라인 설정
      }

      draw() {
        ctx.save();
        ctx.translate(this.x, this.y);
        ctx.rotate(this.rotation);
        ctx.globalAlpha = this.opacity;

        // 1. 키캡의 입체감 있는 3D 베이스 그림자
        ctx.fillStyle = 'rgba(0, 80, 200, 0.35)';
        ctx.beginPath();
        ctx.roundRect(-this.width/2 + 2, -this.height/2 + 3, this.width, this.height, 4);
        ctx.fill();

        // 2. 키캡 메인 바디
        ctx.fillStyle = '#FFFFFF';
        ctx.strokeStyle = '#0076FF';
        ctx.lineWidth = 1.8;
        ctx.beginPath();
        ctx.roundRect(-this.width/2, -this.height/2, this.width, this.height, 4);
        ctx.fill();
        ctx.stroke();

        // 3. 자판 상단 음각 이펙트 느낌 연출용 이중 라인
        ctx.strokeStyle = '#E2E8F0';
        ctx.lineWidth = 1;
        ctx.beginPath();
        ctx.roundRect(-this.width/2 + 2, -this.height/2 + 2, this.width - 4, this.height - 6, 2);
        ctx.stroke();

        // 4. 자판 텍스트 정보 마킹
        ctx.fillStyle = '#0076FF';
        ctx.font = `bold 8px 'Pretendard', sans-serif`;
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
        ctx.fillText(this.label, 0, 1);

        ctx.restore();
      }
    }

    // 다중 "클릭하세요" 버튼 마우스 호버 센싱 및 동역학 물결 시스템 구축
    const clickButtons = document.querySelectorAll('.click-me-btn');

    document.addEventListener('mousemove', (e) => {
      clickButtons.forEach(btn => {
        const rect = btn.getBoundingClientRect();
        
        // 뷰포트 스크롤과 보정값을 연산한 중심좌표 파악
        const btnX = rect.left + rect.width / 2;
        const btnY = rect.top + rect.height / 2;

        const dist = Math.hypot(e.clientX - btnX, e.clientY - btnY);
        const r1 = btn.querySelector('.ripple-ring-1');
        const r2 = btn.querySelector('.ripple-ring-2');

        if (dist < 200) {
          // 마우스 접근 감도(Strength) 연산
          const strength = (200 - dist) / 200;
          r1.style.opacity = strength * 0.9;
          r2.style.opacity = strength * 0.55;
          r1.style.transform = `scale(${1 + strength * 0.35})`;
          r2.style.transform = `scale(${1 + strength * 0.7})`;
          btn.style.transform = `scale(${1 + strength * 0.08})`;
          btn.style.borderColor = `rgba(0, 118, 255, ${0.3 + strength * 0.7})`;
        } else {
          r1.style.opacity = '0';
          r2.style.opacity = '0';
          btn.style.transform = 'scale(1)';
          btn.style.borderColor = 'rgba(226, 232, 240, 1)';
        }
      });
    });

    // 모든 이동 트리거 버튼에 청축 사운드 피드백 및 다음 목적지 페이지 스크롤 전환 바인딩
    clickButtons.forEach(btn => {
      btn.addEventListener('click', (e) => {
        e.stopPropagation();
        playBlueSwitch();

        // 1. 자판 블록 파티클 고밀도 생성
        const clickX = e.clientX;
        const clickY = e.clientY;
        for (let i = 0; i < 20; i++) {
          particles.push(new KeycapParticle(clickX, clickY));
        }

        // 2. 부드러운 스무스 스크롤 네비게이션 트리거
        const targetId = btn.getAttribute('data-next');
        const targetElement = document.querySelector(targetId);
        if (targetElement) {
          setTimeout(() => {
            targetElement.scrollIntoView({ behavior: 'smooth' });
          }, 120);
        }

        showToast("다음 드림 아지트 아일랜드로 발자국을 옮깁니다.");
      });
    });

    // 파티클 상시 물리 갱신 루프 가동
    function animateParticles() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      particles = particles.filter(p => p.opacity > 0);
      particles.forEach(p => {
        p.update();
        p.draw();
      });
      requestAnimationFrame(animateParticles);
    }
    requestAnimationFrame(animateParticles);

    // 전역 기계식 클릭 효과 적용
    document.addEventListener('click', (e) => {
      initAudio();
      if (e.target.closest('a') || e.target.closest('select') || e.target.closest('input') || e.target.closest('textarea')) {
        playBlueSwitch();
      }
    });
  </script>
